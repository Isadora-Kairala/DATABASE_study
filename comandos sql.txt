
--- comando para criar um novo BD :

CREATE DATABASE Aula01


-- comando para entrar no BD que foi criado:
USE Aula01


--comando para criar uma nova tabela:
CREATE TABLE alunos(
  RA int,
  nome varchar(80),
  fone varchar(30),
  data_nasc date
   )

--comando para consultar o conteudo da tabela:
SELECT * FROM alunos

--comando para cadastrar um novo registro na tabela criada:
INSERT INTO alunos
VALUES (123456, 'Ana', '16997969449', '2002/10/02')
